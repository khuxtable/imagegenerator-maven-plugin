41  /**
42   * A custom ComponentConfigurator which adds the project's runtime classpath
43   * elements to the class loader.
44   *
45   * <p>Taken from
46   * http://old.nabble.com/Adding-project-dependencies-and-generated-classes-to-classpath-of-my-plugin-td18624435.html
47   * </p>
48   *
49   * @author             Brian Jackson
50   * @since              Aug 1, 2008 3:04:17 PM
51   * @plexus.component   role="org.codehaus.plexus.component.configurator.ComponentConfigurator"
52   *                     role-hint="include-project-dependencies"
53   * @plexus.requirement role="org.codehaus.plexus.component.configurator.converters.lookup.ConverterLookup"
54   *                     role-hint="default"
55   */
56  public class IncludeProjectDependenciesComponentConfigurator extends AbstractComponentConfigurator {
57  
58      /**
59       * @see org.codehaus.plexus.component.configurator.AbstractComponentConfigurator#configureComponent(java.lang.Object,
60       *      org.codehaus.plexus.configuration.PlexusConfiguration,
61       *      org.codehaus.plexus.component.configurator.expression.ExpressionEvaluator,
62       *      org.codehaus.classworlds.ClassRealm,
63       *      org.codehaus.plexus.component.configurator.ConfigurationListener)
64       */
65      public void configureComponent(Object component, PlexusConfiguration configuration, ExpressionEvaluator expressionEvaluator,
66              ClassRealm containerRealm, ConfigurationListener listener) throws ComponentConfigurationException {
67          addProjectDependenciesToClassRealm(expressionEvaluator, containerRealm);
68  
69          converterLookup.registerConverter(new ClassRealmConverter(containerRealm));
70  
71          ObjectWithFieldsConverter converter = new ObjectWithFieldsConverter();
72  
73          converter.processConfiguration(converterLookup, component, containerRealm.getClassLoader(), configuration,
74                                         expressionEvaluator, listener);
75      }
76  
77      /**
78       * Add the project dependencies to the class realm so that they are
79       * accessible.
80       *
81       * @param  expressionEvaluator the expression evaluator.
82       * @param  containerRealm      the container realm.
83       *
84       * @throws ComponentConfigurationException if an error occurs.
85       */
86      @SuppressWarnings("unchecked")
87      private void addProjectDependenciesToClassRealm(ExpressionEvaluator expressionEvaluator, ClassRealm containerRealm)
88          throws ComponentConfigurationException {
89          List<String> runtimeClasspathElements;
90  
91          try {
92              // noinspection unchecked
93              runtimeClasspathElements = (List<String>) expressionEvaluator.evaluate("${project.runtimeClasspathElements}");
94          } catch (ExpressionEvaluationException e) {
95              throw new ComponentConfigurationException("There was a problem evaluating: ${project.runtimeClasspathElements}", e);
96          }
97  
98          // Add the project dependencies to the ClassRealm
99          final URL[] urls = buildURLs(runtimeClasspathElements);
100 
101         for (URL url : urls) {
102             containerRealm.addConstituent(url);
103         }
104     }
105 
106     /**
107      * Build an array of URLs representing the project dependencies.
108      *
109      * @param  runtimeClasspathElements a list of Strings of runtime classpath
110      *                                  elements.
111      *
112      * @return an array of URLs for the classpath elements.
113      *
114      * @throws ComponentConfigurationException if an error occurs.
115      */
116     private URL[] buildURLs(List<String> runtimeClasspathElements) throws ComponentConfigurationException {
117         // Add the projects classes and dependencies
118         List<URL> urls = new ArrayList<URL>(runtimeClasspathElements.size());
119 
120         for (String element : runtimeClasspathElements) {
121             try {
122                 final URL url = new File(element).toURI().toURL();
123 
124                 urls.add(url);
125             } catch (MalformedURLException e) {
126                 throw new ComponentConfigurationException("Unable to access project dependency: " + element, e);
127             }
128         }
129 
130         // Add the plugin's dependencies (so Trove stuff works if Trove isn't on
131         return urls.toArray(new URL[urls.size()]);
132     }
